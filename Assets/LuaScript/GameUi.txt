GameUi = {}
local unity = CS.UnityEngine
tag = "uiComponent"
-- local canvas = unity.GameObject.Find("Canvas").transform
print("GameUi Run")
function GameUi:LoadUi(ui)
	obj = CS.ResManager.LoadPrefab(ui.path)
	local panel = unity.GameObject.Instantiate(obj)
	-- panel.transform:SetParent(canvas,false)
	children = panel.transform:GetComponentsInChildren(typeof(unity.Transform),true)
	print(children)
	for i = 0, children.Length - 1 do
    	local child = children[i]
    	print(child.tag)
    	if child.tag == tag then 
    		ui:SetUiComponent(child)
    	end
	end
end
